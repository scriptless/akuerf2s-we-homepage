Auf Deno ist TypeScript ohne Übersetzung direkt ablauffähig. Deno ist also die Runtime Engine für TypeScript. Schreiben Sie in TypeScript möglichst umfangreich statisch typisiert ein auf Deno ablauffähiges Programm server.ts, das Sie mit deno run --allow-net server.ts starten können, das die COVID-19-Fallzahlen in Deutschland einliest, Minimum, Maximum, Durchschnitt und Summe berechnet und diese auf einer Webseite ausgibt. Ein Beispiel-Programm zu server.ts finden Sie im Deno-Manual unter Making an HTTP request

async function serveHttp(conn: Deno.Conn) {
  const httpConn = Deno.serveHttp(conn);
  for await (const requestEvent of httpConn) {
	  
	const lowest: Bundesland = data.sort((a: Bundesland, b: Bundesland) => a.anzahl - b.anzahl)[0]
	const highest: Bundesland = data.sort((a: Bundesland, b: Bundesland) => b.anzahl - a.anzahl)[0]
	const summe: number = data.filter((a: Bundesland) => a.anzahl).reduce((a: number,x: Bundesland) => a + x.anzahl, 0)
	const durchschnitt: number = summe / data.length
    const body = `wenigsten Fälle: ${lowest.bundesland} mit ${lowest.anzahl} Fällen\nmeisten Fälle: ${highest.bundesland} mit ${highest.anzahl} Fällen\nDurschnitt Fälle aller Bundesländer: ${durchschnitt}\nGesamt Fälle Deutschland: ${summe}`;

    requestEvent.respondWith(
      new Response(body, {
        status: 200,
      }),
    );
	
  }
  
}
